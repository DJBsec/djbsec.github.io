import os
import re
import asyncio
import edge_tts
from datetime import datetime

# ================================

# 📂 PATHS

# ================================

INPUT_SUMMARIES_FILE = "/Users/ben/Documents/GitHub/CyberNews/finalize_daily_news.py"
POSTS_FOLDER = "/Users/ben/Documents/GitHub/djbsec.github.io/\_posts"
AUDIO_FOLDER = "/Users/ben/Documents/GitHub/djbsec.github.io/assets/audio/news"

# ================================

# 🔊 GENERATE AUDIO USING EDGE_TTS

# ================================

async def generate_audio(text, output_file):
print("Generating audio file...")
communicate = edge_tts.Communicate(text, "en-US-GuyNeural")
await communicate.save(output_file)
print(f"Audio file saved: {output_file}")

# ================================

# 🚀 MAIN RUNNER

# ================================

def main():
today = datetime.now().strftime("%Y-%m-%d")
date_nodash = datetime.now().strftime("%m%d")
audio_filename = f"CyberSecurityNews-{today}.mp3"
audio_path = os.path.join(AUDIO_FOLDER, audio_filename)

    # Read summaries from ChatGPT output file
    if not os.path.exists(INPUT_SUMMARIES_FILE):
        print(f"Summaries file not found: {INPUT_SUMMARIES_FILE}")
        return

    with open(INPUT_SUMMARIES_FILE, "r", encoding="utf-8") as f:
        summaries = f.read()

    # Remove markdown URLs for audio
    summaries_no_urls = re.sub(r'\[.*?\]\(.*?\)', '', summaries)

    # Remove markdown heading symbols (#) for audio clarity
    summaries_no_headings = re.sub(r'^\s*#+\s*', '', summaries_no_urls, flags=re.MULTILINE)

    # Generate TTS intro + summaries (without URLs and headings)
    full_text = f"Welcome to the CyberSecurity News for {datetime.now().strftime('%B %d, %Y')}. Here are today's top cybersecurity headlines.\n\n{summaries_no_headings}"

    # Generate audio
    asyncio.run(generate_audio(full_text, audio_path))

    # Build markdown content (keeps original formatting with URLs and headings)
    md_content = f"""---

layout: post
title: DJBSEC's CyberNews {today}
date: {today}
description: {today}
tags: CyberNews
categories: News
thumbnail: assets/img/cybernews.webp
featured: false

---

<div class="row mt-3">
    <div class="col-sm mt-3 mt-md-0">
        {{% include audio.liquid path="assets/audio/news/{audio_filename}" controls=true %}}
    </div>
</div>

{summaries}
"""

    # Write markdown post
    md_filename = f"{today}-cybernews{date_nodash}.md"
    md_path = os.path.join(POSTS_FOLDER, md_filename)
    with open(md_path, "w", encoding="utf-8") as f:
        f.write(md_content)

    print(f"Markdown post created: {md_path}")

# ================================

# 🔧 RUN SCRIPT

# ================================

if **name** == "**main**":
main()
